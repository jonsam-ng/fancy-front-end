  function loadApps() {

  function performAppChanges() {
    return Promise.resolve().then(() => {
      // https://github.com/single-spa/single-spa/issues/545
      window.dispatchEvent(
        new CustomEvent(
          appsThatChanged.length === 0

            "single-spa:before-mount-routing-event",

        .map(toUnmountPromise)
        .map((unmountPromise) => unmountPromise.then(toUnloadPromise));

      // ç­‰å¾… unload å®Œæˆçš„ promises

      unmountAllPromise.then(() => {
        // unload å®Œæˆåï¼Œè§¦å‘ before-mount-routing-event äº‹ä»¶

          new CustomEvent(
            "single-spa:before-mount-routing-event",
            getCustomEventDetail(true)
          )
        );
      });

      /* We load and bootstrap apps while other apps are unmounting, but we
       * wait to mount the app until all apps are finishing unmounting
       */
      // åœ¨ä¸Šè¿°åº”ç”¨ unmounting çš„è¿‡ç¨‹ä¸­å…ˆå°è¯•å°† appsToLoad ä¸­çš„åº”ç”¨è¿›è¡Œ bootstrap å’Œ load

      const loadThenMountPromises = appsToLoad.map((app) => {
        return toLoadPromise(app).then((app) =>
          tryToBootstrapAndMount(app, unmountAllPromise)
        );
      });

      /* These are the apps that are already bootstrapped and just need
       * to be mounted. They each wait for all unmounting apps to finish up
       * before they mount.
       */
      // è¿‡æ»¤æ‰å·²ç»åœ¨ appsToLoad å¤„ç†è¿‡çš„çš„åº”ç”¨ï¼Œåœ¨è°ƒç”¨ tryToBootstrapAndMount

      const mountPromises = appsToMount
        .filter((appToMount) => appsToLoad.indexOf(appToMount) < 0)
        .map((appToMount) => {
          return tryToBootstrapAndMount(appToMount, unmountAllPromise);
        });

      return unmountAllPromise
        .catch((err) => {

          callAllEventListeners();

          /* Now that the apps that needed to be unmounted are unmounted, their DOM navigation
           * events (like hashchange or popstate) should have been cleaned up. So it's safe
           * to let the remaining captured event listeners to handle about the DOM event.
           */
          // éœ€è¦è¢« unmount çš„åº”ç”¨éƒ½å·²ç» unmount

  // CustomEvent ä½¿ç”¨äº† https://github.com/webmodules/custom-event/blob/master/index.js
  function getCustomEventDetail(isBeforeChanges = false, extraProperties) {
    const newAppStatuses = {};
    const appsByNewStatus = {
      // for apps that were mounted
      [MOUNTED]: [],
      // for apps that were unmounted
      [NOT_MOUNTED]: [],
      // apps that were forcibly unloaded

      // apps that attempted to do something but are broken now

        addApp(app, MOUNTED);
      });

        addApp(app, NOT_LOADED);
      });

        addApp(app, NOT_MOUNTED);

        newAppStatuses,
        appsByNewStatus,
        totalAppChanges: appsThatChanged.length,
        originalEvent: eventArguments?.[0],

        newUrl,

      },
    };

    function addApp(app, status) {

        appsByNewStatus[status] || []);

      statusArr.push(appName);

  }

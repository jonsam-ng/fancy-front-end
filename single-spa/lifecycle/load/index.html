export function toUnloadPromise(app) {

       * anything that still thinks we need to unload the app.
       */

      return app;
    }
    // å¯èƒ½æ˜¯ unloadApplication å’Œ reroute åŒæ—¶æƒ³è¦ unload åº”ç”¨
    // ç­‰å¾…å·²ç» unload æ‰§è¡Œå®Œæ¯•
    if (app.status === UNLOADING) {
      /* Both unloadApplication and reroute want to unload this app.
       * It only needs to be done once, though.
       */
      return unloadInfo.promise.then(() => app);
    }
    // å¦‚æœåº”ç”¨çŠ¶æ€ä¸æ˜¯ NOT_MOUNTED æˆ–è€… LOAD_ERRORï¼Œæ²¡æœ‰åŠæ³• unloadï¼Œç›´æ¥è¿”å›
    // éœ€è¦å…ˆ unmount åˆ° NOT_MOUNTED çŠ¶æ€æ‰èƒ½ unload

       */
      return app;
